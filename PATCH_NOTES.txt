How to wire into your page (index.tsx) — minimal edits:
1) After you compute 'enrichedAll', create a proximity-sorted array:
   const proximitySorted = useMemo(() => [...enrichedAll].sort((a:any,b:any)=>(a.distanceKm||0)-(b.distanceKm||0)), [enrichedAll]);

2) Select the final 200 with caps:
   import { selectTopK } from "@/lib/selectTopK";
   import { makeSelectionConfig } from "@/lib/categoryCaps";
   import { getPrimaryCategory } from "@/components/categoryMeta";
   import { isFastFoodFromTypes } from "@/lib/isFastFood";

   const selectedPool = useMemo(() => {
     const cfg = makeSelectionConfig<any>({
       getCategory: (p) => getPrimaryCategory(Array.isArray(p.types)?p.types: (typeof p.types==="string"?[p.types]:[])),
       getTypes: (p) => (Array.isArray(p.types)?p.types: (typeof p.types==="string"?[p.types]:[])),
       isFastFood: (p) => isFastFoodFromTypes(Array.isArray(p.types)?p.types: (typeof p.types==="string"?[p.types]:[])),
     });
     return selectTopK<any>(proximitySorted, cfg);
   }, [proximitySorted]);

3) From that point on, use 'selectedPool' instead of 'enrichedAll' for filtering/sorting:
   const filtered = useMemo(() => {
     if (selectedCategory === "All") return selectedPool;
     const wanted = new Set((CATEGORY_MAP[selectedCategory] || []).map((s) => s.toLowerCase()));
     return selectedPool.filter((p:any) => (p.types || []).some((t:string) => wanted.has(t.toLowerCase())));
   }, [selectedPool, selectedCategory]);
